136 template <std::size_t Capacity>
137 constexpr std::size_t RingWrap(const std::size_t ring_index) {
138  // This is a bit faster than `return ring_index % Capacity` (~40% reduction in
139  // Speed.PushBackOverFull test)
140  return (ring_index <= Capacity) ? ring_index : ring_index - Capacity - 1;
141 }

269 template <typename Ptr,
270  typename AllocTraits,
271  std::size_t Capacity,
272  typename OutputIt>
273 OutputIt copy(const Iterator<Ptr, AllocTraits, Capacity>& begin,
274  const Iterator<Ptr, AllocTraits, Capacity>& end,
275  OutputIt out) {
276  assert(begin <= end);
277 
278  if (begin == end) {
279  // Empty range, pass
280  } else if (&*end > &*begin) {
281  // Fully contiguous range.
282  out = std::copy(&*begin, &*end, out);
283  } else {
284  // Copy in two sections.
285  out = std::copy(&*begin, &begin.data_[Capacity + 1], out);
286  out = std::copy(end.data_, &*end, out);
287  }
288 
289  return out;
290 }

305 template <typename Elem,
306  std::size_t Capacity,
307  typename Allocator = std::allocator<Elem>>
308 class RingBuf {
309  public:
310  using allocator_type = Allocator;
311  using alloc_traits = std::allocator_traits<allocator_type>;
312  using value_type = Elem;
313  using pointer = typename alloc_traits::pointer;
314  using const_pointer = typename alloc_traits::const_pointer;
315  using reference = decltype(*pointer{});
316  using const_reference = decltype(*const_pointer{});
317  using iterator = detail::ringbuf::Iterator<pointer, alloc_traits, Capacity>;
318  using const_iterator =
319  detail::ringbuf::Iterator<const_pointer, alloc_traits, Capacity>;
320  using reverse_iterator = std::reverse_iterator<iterator>;
321  using const_reverse_iterator = std::reverse_iterator<const_iterator>;
322  using difference_type = typename alloc_traits::difference_type;
323  using size_type = typename alloc_traits::size_type;
324 
325  using self = RingBuf<Elem, Capacity>;
326 
327  private:
328  // The allocator is used to allocate memory, and to construct and destroy
329  // elements.
330  allocator_type alloc_{};
331 
332  // The start of the dynamically allocated backing array.
333  pointer data_{nullptr};
334  // The next position to write to for push_back().
335  size_type next_{0U};
336 
337  // Start of the ring buffer in data_.
338  size_type ring_offset_{0U};
339  // The number of elements in the ring buffer (distance between begin() and
340  // end()).
341  size_type size_{0U};
342 
343  constexpr static size_type Decrement(const size_type index) {
344  return index > 0 ? index - 1 : Capacity;
345  }
346 
347  constexpr static size_type Increment(const size_type index) {
348  return index < (Capacity) ? index + 1 : 0;
349  }
350 
351  // Swap everything but the allocator - caller has to figure that out
352  // separately.
353  void Swap(RingBuf& other) noexcept {
354  std::swap(data_, other.data_);
355  std::swap(next_, other.next_);
356  std::swap(ring_offset_, other.ring_offset_);
357  std::swap(size_, other.size_);
358  }
359 
360  // Move things after pop_front.
361  void ShrinkFront() noexcept {
362  ring_offset_ = Increment(ring_offset_);
363  // Precondition: size != 0 (when it is, pop_front returns early.)
364  size_--;
365  }
366 
367  // Move things around before pop_back destroys the last entry.
368  void ShrinkBack() noexcept {
369  next_ = Decrement(next_);
370  // Precondition: size != 0 (when it is, pop_back returns early.)
371  size_--;
372  }
373 
374  // Move things around before emplace_front constructs its new entry.
375  void GrowFront() noexcept {
376  // Move ring_offset_ down, and possibly around
377  ring_offset_ = Decrement(ring_offset_);
378  // Precondition: size != Capacity (when it is, emplace_front pop_backs
379  // first.)
380  size_++;
381  }
382 
383  // Move things around after emplace_back.
384  void GrowBack() noexcept {
385  next_ = Increment(next_);
386  // Precondition: size != Capacity (when it is, emplace_back pop_fronts
387  // first)
388  size_++;
389  }
390 
391  public:
399  RingBuf() : RingBuf(allocator_type{}){};

410  : alloc_(allocator),
411  data_(alloc_traits::allocate(alloc_, Capacity + 1)) {}

445

506  alloc_traits::propagate_on_container_move_assignment::value ||

508  if (alloc_traits::propagate_on_container_move_assignment::value ||

693  template <typename... Args>
694  void emplace_front(Args&&... args) {
695  if (max_size() == 0) {
696  // A buffer of size zero is conceptually sound, so let's support it.
697  return;
698  }
699 
700  alloc_traits::construct(alloc_, &data_[Decrement(ring_offset_)],
701  std::forward<Args>(args)...);
702 
703  // If required, make room for next time.
704  if (size() == max_size()) {
705  pop_back();
706  }
707  GrowFront();
708  }
709 
715  void push_back(const_reference value) { return emplace_back(value); }
721  void push_back(value_type&& value) { return emplace_back(std::move(value)); }
729  template <typename... Args>
730  void emplace_back(Args&&... args) {
731  if (max_size() == 0) {
732  // A buffer of size zero is conceptually sound, so let's support it.
733  return;
734  }
735 
736  alloc_traits::construct(alloc_, &data_[next_], std::forward<Args>(args)...);
737 
738  // If required, make room for next time.
739  if (size() == max_size()) {
740  pop_front();
741  }
742  GrowBack();
743  }
744 
749  void pop_front() noexcept {
750  if (size() == 0) {
751  return;
752  }
753 
754  alloc_traits::destroy(alloc_, &data_[ring_offset_]);
755  ShrinkFront();
756  }
761  void pop_back() noexcept {
762  if (size() == 0) {
763  return;
764  }
765 
766  ShrinkBack();
767  alloc_traits::destroy(alloc_, &data_[next_]);
768  }
775  void clear() noexcept(noexcept(pop_front())) {
776  // It might be fractionally more efficient to iterate through begin..end and
777  // allocator::destroy each one, but this is a lot nicer to read.
778  while (!empty()) {
779  pop_front();
780  }
781  }
