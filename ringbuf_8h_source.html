136 template <std::size_t Capacity>
137 constexpr std::size_t RingWrap(const std::size_t ring_index) {
138  // This is a bit faster than `return ring_index % Capacity` (~40% reduction in
139  // Speed.PushBackOverFull test)
140  return (ring_index <= Capacity) ? ring_index : ring_index - Capacity - 1;
141 }

299 template <typename Ptr,
300  typename AllocTraits,
301  std::size_t Capacity,
302  typename OutputIt>
303 OutputIt copy(const Iterator<Ptr, AllocTraits, Capacity>& begin,
304  const Iterator<Ptr, AllocTraits, Capacity>& end,
305  OutputIt out) {
306  assert(begin <= end);
307 
308  if (begin == end) {
309  // Empty range, pass
310  } else if (&*end > &*begin) {
311  // Fully contiguous range.
312  out = std::copy(&*begin, &*end, out);
313  } else {
314  // Copy in two sections.
315  out = std::copy(&*begin, &begin.data_[Capacity + 1], out);
316  out = std::copy(end.data_, &*end, out);
317  }
318 
319  return out;
320 }

335 template <typename Elem,
336  std::size_t Capacity,
337  typename Allocator = std::allocator<Elem>>
338 class RingBuf {
339  public:
340  using allocator_type = Allocator;
341  using alloc_traits = std::allocator_traits<allocator_type>;
342  using value_type = Elem;
343  using pointer = typename alloc_traits::pointer;
344  using const_pointer = typename alloc_traits::const_pointer;
345  using reference = decltype(*pointer{});
346  using const_reference = decltype(*const_pointer{});
347  using iterator = detail::ringbuf::Iterator<pointer, alloc_traits, Capacity>;
348  using const_iterator =
349  detail::ringbuf::Iterator<const_pointer, alloc_traits, Capacity>;
350  using reverse_iterator = std::reverse_iterator<iterator>;
351  using const_reverse_iterator = std::reverse_iterator<const_iterator>;
352  using difference_type = typename alloc_traits::difference_type;
353  using size_type = typename alloc_traits::size_type;
354 
355  using self = RingBuf<Elem, Capacity>;
356 
357  private:
358  // The allocator is used to allocate memory, and to construct and destroy
359  // elements.
360  allocator_type alloc_{};
361 
362  // The start of the dynamically allocated backing array.
363  pointer data_{nullptr};
364  // The next position to write to for push_back().
365  size_type next_{0U};
366 
367  // Start of the ring buffer in data_.
368  size_type ring_offset_{0U};
369  // The number of elements in the ring buffer (distance between begin() and
370  // end()).
371  size_type size_{0U};
372 
373  constexpr static size_type Decrement(const size_type index) {
374  return index > 0 ? index - 1 : Capacity;
375  }
376 
377  constexpr static size_type Increment(const size_type index) {
378  return index < (Capacity) ? index + 1 : 0;
379  }
380 
381  // Swap everything but the allocator - caller has to figure that out
382  // separately.
383  void Swap(RingBuf& other) noexcept {
384  std::swap(data_, other.data_);
385  std::swap(next_, other.next_);
386  std::swap(ring_offset_, other.ring_offset_);
387  std::swap(size_, other.size_);
388  }
389 
390  // Move things after pop_front.
391  void ShrinkFront() noexcept {
392  ring_offset_ = Increment(ring_offset_);
393  // Precondition: size != 0 (when it is, pop_front returns early.)
394  size_--;
395  }
396 
397  // Move things around before pop_back destroys the last entry.
398  void ShrinkBack() noexcept {
399  next_ = Decrement(next_);
400  // Precondition: size != 0 (when it is, pop_back returns early.)
401  size_--;
402  }
403 
404  // Move things around before emplace_front constructs its new entry.
405  void GrowFront() noexcept {
406  // Move ring_offset_ down, and possibly around
407  ring_offset_ = Decrement(ring_offset_);
408  // Precondition: size != Capacity (when it is, emplace_front pop_backs
409  // first.)
410  size_++;
411  }
412 
413  // Move things around after emplace_back.
414  void GrowBack() noexcept {
415  next_ = Increment(next_);
416  // Precondition: size != Capacity (when it is, emplace_back pop_fronts
417  // first)
418  size_++;
419  }
420 
421  public:
429  RingBuf() : RingBuf(allocator_type{}){};
439  explicit RingBuf(const allocator_type& allocator)
440  : alloc_(allocator),
441  data_(alloc_traits::allocate(alloc_, Capacity + 1)) {}
442 
448  ~RingBuf() {
449  clear();
450  alloc_traits::deallocate(alloc_, data_, Capacity + 1);
451  }

472  RingBuf(const RingBuf& other, const allocator_type& allocator)

486  Swap(other);
487  }

499  if (other.alloc_ == allocator) {

501  } else {
502  for (auto& element : other) {
503  emplace_back(std::move(element));
504  }

517  clear();

535  RingBuf& operator=(RingBuf&& other) noexcept(
536  alloc_traits::propagate_on_container_move_assignment::value ||
537  std::is_nothrow_move_constructible<value_type>::value) {
538  if (alloc_traits::propagate_on_container_move_assignment::value ||
539  alloc_ == other.alloc_) {
540  // We're either getting the other's allocator or they're already the same,
541  // so swap data in one go.
542  detail::ringbuf::MoveAllocator(alloc_, other.alloc_);
543  Swap(other);
544  } else {
545  // Different allocators and can't swap them, so move elementwise.
546  clear();
547  for (auto& element : other) {
548  emplace_back(std::move(element));
549  }
550  }
551

553  }

723  template <typename... Args>
724  void emplace_front(Args&&... args) {
725  if (max_size() == 0) {
726  // A buffer of size zero is conceptually sound, so let's support it.
727  return;
728  }
729 
730  alloc_traits::construct(alloc_, &data_[Decrement(ring_offset_)],
731  std::forward<Args>(args)...);
732 
733  // If required, make room for next time.
734  if (size() == max_size()) {
735  pop_back();
736  }
737  GrowFront();
738  }
739 
745  void push_back(const_reference value) { return emplace_back(value); }
751  void push_back(value_type&& value) { return emplace_back(std::move(value)); }
759  template <typename... Args>
760  void emplace_back(Args&&... args) {
761  if (max_size() == 0) {
762  // A buffer of size zero is conceptually sound, so let's support it.
763  return;
764  }
765 
766  alloc_traits::construct(alloc_, &data_[next_], std::forward<Args>(args)...);
767 
768  // If required, make room for next time.
769  if (size() == max_size()) {
770  pop_front();
771  }
772  GrowBack();
773  }
774 
779  void pop_front() noexcept {
780  if (size() == 0) {
781  return;
782  }
783 
784  alloc_traits::destroy(alloc_, &data_[ring_offset_]);
785  ShrinkFront();
786  }
791  void pop_back() noexcept {
792  if (size() == 0) {
793  return;
794  }
795 
796  ShrinkBack();
797  alloc_traits::destroy(alloc_, &data_[next_]);
798  }
805  void clear() noexcept(noexcept(pop_front())) {
806  // It might be fractionally more efficient to iterate through begin..end and
807  // allocator::destroy each one, but this is a lot nicer to read.
808  while (!empty()) {
809  pop_front();
810  }
811  }
